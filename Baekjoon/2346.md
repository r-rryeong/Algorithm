### 풍선 터뜨리기

```python
import sys
from collections import deque

n = int(input())
b = deque(i for i in range(1, n+1))   # 풍선 번호
num = deque(map(int, input().split()))   # 종이에 적힌 번호
print(b[0], end=' ')
cnt = num[0]
while len(num)!=1:
    b.popleft()
    num.popleft()
    if cnt > 0:   # 양수일 경우 오른쪽으로 이동
        while cnt!=1:
            x = num.popleft()
            num.append(x)
            y = b.popleft()
            b.append(y)
            cnt -= 1
    else:   # 음수일 경우 왼쪽으로 이동
        while cnt!=0:
            x = num.pop()
            num.appendleft(x)
            y = b.pop()
            b.appendleft(y)
            cnt += 1
    print(b[0], end=' ')
    cnt = num[0]
```
