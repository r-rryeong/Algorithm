### 최댓값

```python
max_num, r, c = -1, 0, 0
lst = []
for _ in range(9):
    l = list(map(int, input().split()))
    lst.append(l)

for i in range(9):
    for j in range(9):
        if lst[i][j] == 99:
            max_num, r, c = 99, i+1, j+1
            break

        if lst[i][j] > max_num:
            max_num, r, c = lst[i][j], i+1, j+1

    if max_num == 99:
        break

print(max_num)
print(r, c)
```
