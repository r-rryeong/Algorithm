```python
# 1. for문

# 점화식 구하기!
# 점화식: D[N] = min(D[N/3]+1, D[N/2]+1, D[N-1]+1)

N = int(input())
d = [0] * (N+1)

for i in range(2, N+1):
    d[i] = d[i-1] + 1    # 2의 배수도 아니고 3의 배수도 아닌 경우는 해당 값을 가짐
    # d[0]=0, d[1]=1
    if i % 2 == 0:    # i가 2로 나누어 떨어지는 경우
        # 최소 연산 횟수를 구하기 때문에 min 사용
        d[i] = min(d[i], d[i // 2] + 1)
    if i % 3 == 0:    # i가 3으로 나누어 떨어지는 경우
        d[i] = min(d[i], d[i // 3] + 1)

print(d[N])
```
