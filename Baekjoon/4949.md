### 균형잡힌 세상

```python
import sys
input = sys.stdin.readline

while True:
    string = input().rstrip()

    # 입력의 종료조건을 만나면 while문 종료
    if string == '.':
        break

    stack = []
    flag = True    # 짝이 맞는지 나타낼 변수(default는 True)

    # 문자열 탐색
    for s in string:
        # 여는 괄호는 모두 스택에 추가
        if s == '(' or s == '[':
            stack.append(s)
        # 닫는 소괄호를 만난 경우
        elif s == ')':
            if not stack or stack[-1]=='[':    # 스택이 비어있거나 짝이 맞지 않는 경우
                flag = False                   # flag를 False로 변경
                break                          # for문 종료
            elif stack[-1]=='(':          # 짝이 맞으면
                stack.pop()               # pop
        # 닫는 대괄호를 만난 경우
        elif s == ']':
            if not stack or stack[-1]=='(':
                flag = False
                break
            elif stack[-1]=='[':
                stack.pop()

    if flag==True and not stack:    # flag가 True이고 스택이 비어있으면
        print('yes')                # 짝이 다 맞음
    else:
        print('no')
```
