### 222-풀링

```python
import sys
input = sys.stdin.readline

N = int(input())
arr = [list(map(int, input().split())) for _ in range(N)]
while len(arr[-1]) != 1:
    lst = []
    for i in range(0, len(arr), 2):
        block = []
        for j in range(0, len(arr), 2):
            tmp = [arr[i][j], arr[i][j+1], arr[i+1][j], arr[i+1][j+1]]
            tmp.sort()
            block.append(tmp[-2])
        lst.append(block)
    arr = lst
print(arr[0][0])
```
