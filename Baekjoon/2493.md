```python
import sys
input = sys.stdin.readline

N = int(input())
if N==1:
    print(0)
    exit()
tower = list(map(int, input().split()))
stack = []
for i in range(N):
    if not stack:
        stack.append([tower[i], i])
        print(0, end=" ")
    else:
        while stack:
            if stack[-1][0] < tower[i]:
                stack.pop()
                if not stack:
                    stack.append([tower[i], i])
                    print(0, end=" ")
                    break
            elif stack[-1][0] >= tower[i]:
                print(stack[-1][1]+1, end=" ")
                stack.append([tower[i], i])
                break
```
