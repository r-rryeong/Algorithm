### 호석이 두 마리 치킨

```python
# 5/19

import sys
input = sys.stdin.readline

def dfs(a, b):

    check_lst[a] = 1
    check_lst[b] = 1

    dis = 0
    for i in range(1, N+1):
        for j in range(1, N+1):
            if graph[i][j]:
                if not check_lst[j]:
                    check_lst[j] = 1
                    dis += 1
                    if j == a or j == b:
                        break

    return dis

N, M = map(int, input().split())
graph = [[0]*(N+1) for _ in range(N+1)]
for _ in range(M):
    a, b = map(int, input().split())
    graph[a][b] = 1
    graph[b][a] = 1

store = []   # 가능한 건물
min_dis = sys.maxsize
for i in range(1, N+1):
    for j in range(i+1, N+1):
        check_lst = [0] * (N + 1)
        cnt = dfs(i, j)

        if cnt < min_dis:
            min_dis = cnt

            store.append([i, j])

store.sort(key=lambda x :(x[0], x[1]))
print(*store[0], min_dis*2)
```
