### 영재의 시험

```python
def solution(depth, cnt):
    global n
    if depth == 10:
        n += 1   # 경우의 수 +1
        return

    for i in range(1, 6):   # 5지 선다
        # 연속하는 3문제의 답이 같지 않도록
        if len(number) < 2 or number[-2] != number[-1] or number[-1] != i:
            number.append(i)
            if answer[len(number)-1] == i:   # 정답인 경우
                solution(depth+1, cnt+1)
            else:    # 틀린 경우
                # 남은 문제를 다 맞춰도 5점이 안되는 경우
                if 10-len(number)+cnt < 5:
                    number.pop()
                    continue
                solution(depth+1, cnt)
            number.pop()

answer = list(map(int, input().split()))
number = []
n = 0
solution(0, 0)
print(n)
```
