### 약수들의 합

```python
def check(n):
    res = False
    lst = []
    for i in range(1, n//2+1):
        if n % i == 0:
            lst.append(i)

    if sum(lst) == n:
        res = True

    return res, lst

while True:
    n = int(input())
    if n < 0:
        break
    res, lst = check(n)
    if res:
        print(str(n) + " = ", end="")
        for idx in range(len(lst)):
            if idx == len(lst)-1:
                print(str(lst[idx]))
            else:
                print(str(lst[idx]) + ' + ', end="")
    else:
        print(n, "is NOT perfect.")
```
