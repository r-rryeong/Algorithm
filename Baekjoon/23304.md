### 아카라카

```python
import sys
input = sys.stdin.readline

def pelindrom(s):
    if s == s[::-1]:
        return True
    else:
        return False

def solution(S):
    global answer
    if len(S) == 1:
        return S

    m = len(S)//2
    if len(S) % 2 == 0:
        l = S[:m]
        r = S[m:]
    else:
        l = S[:m]
        r = S[m+1:]

    if l != r:
        answer = "IPSELENTI"
    elif not pelindrom(S):
        answer = "IPSELENTI"
    elif not pelindrom(solution(l)):
        answer = "IPSELENTI"
    return S

S = input().rstrip()
answer = "AKARAKA"
solution(S)
print(answer)
```
