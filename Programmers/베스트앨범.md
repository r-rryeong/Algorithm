def solution(genres, plays):
    answer = []

    play_dict = dict()   # key: 장르, value: [고유번호, 재생 횟수]
    for idx, cnt in enumerate(plays):
        if genres[idx] not in play_dict:   # 새로 추가하는 장르의 경우(리스트 생성)
            play_dict[genres[idx]] = [[idx, cnt]]
        else:   # 이미 있는 장르의 경우(리스트에 추가)
            play_dict[genres[idx]].append([idx, cnt])

    # 각 장르 재생횟수의 총합
    sum_play = []   # [장르, 총합]
    for play_lst in play_dict:
        s = 0   # 총합
        for p_l in play_dict[play_lst]:
            s += p_l[1]
        sum_play.append([play_lst, s])
    sum_play.sort(key=lambda x: -x[1])   # 재생횟수의 총합이 높은 순으로 정렬
    # 앨범에 들어갈 장르 순서
    genre_lst = []
    for genre in sum_play:
        genre_lst.append(genre[0])

    for genre in genre_lst:
        c = 0   # 추가 횟수
        play_dict[genre].sort(key=lambda x: (x[1], -x[0]))   # 재생 횟수가 낮은 순 -> 고유번호가 높은 순으로 정렬
        while c != 2 and len(play_dict[genre]) != 0:   # 장르 내에서 2개를 추가하지 않았고 리스트가 존재하는 경우
            p = play_dict[genre][-1][0]   # 고유번호
            answer.append(p)
            play_dict[genre].pop()   # 고유번호를 추가한 노래 리스트는 pop
            c += 1

    return answer

def solution(genres, plays):
    answer = []
    
    genre_dict = {}   # {장르: 고유번호}
    for i in range(len(genres)):
        if genre_dict.get(genres[i]):
            genre_dict[genres[i]].append(i)
        else:
            genre_dict[genres[i]] = [i]
    
    # 1번 조건(속한 노래가 많이 재생된 장르)
    play_sum = []   # [장르, 총 재생 횟수]
    for genre in genre_dict.keys():
        s = 0
        for idx in genre_dict[genre]:
            s += plays[idx]
        play_sum.append([genre, s])
        
    play_sum.sort(reverse=True, key=lambda x: x[1])   # 재생 횟수가 큰 장르 기준으로 내림차순 정렬
    
    # 2번 조건(장르 내에서 많이 재생된 노래)
    for genre, _ in play_sum:
        lst = []   # [고유번호, 재생 횟수]
        for idx in genre_dict[genre]:
            lst.append([idx, plays[idx]])
        lst.sort(reverse=True, key=lambda x:x[1])   # 재생 횟수가 큰 번호를 기준으로 내림차순 정렬(고유번호는 오름차순으로 됨)
        
        l = 2   # 장르 별로 최대 두 개씩 수록
        if len(lst) < 2:
            l = len(lst)
            
        for i in range(l):
            idx, cnt = lst.pop(0)
            answer.append(idx)
    
    return answer
