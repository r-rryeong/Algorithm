- 캐시 교체 알고리즘은 `LRU`(Least Recently Used)사용

```python
def solution(cacheSize, cities):
    answer = 0
    q = []
    
    if cacheSize==0:
        answer = len(cities) * 5
    else:
        for c in cities:
            c = c.lower()    # 도시이름을 소문자로 변환(같은 글자를 찾기위해)
            if c in q:       # 캐시에 있는 것이 들어온 경우
                q.remove(c)   # c의 상태를
                q.append(c)   # 제일 최신으로 변경
                answer += 1
            else:            # 캐시에 없는 것이 들어온 경우
                if len(q) >= cacheSize:   # 캐시가 이미 다찼으면
                    q.pop(0)     # 가장 오래된 것을 지우고
                q.append(c)      # 제일 뒤에 추가
                answer += 5
        
    return answer
```

```python
def solution(cacheSize, cities):
    answer = 0    
    cache = []
    
    if cacheSize == 0:   # 캐시크기가 0인 경우
        answer = 5 * len(cities)
        
        return answer
    
    for i in range(len(cities)):
        c = cities[i].lower()   # 대소문자 구분 없이
        # cache hit
        if c in cache:
            cache.remove(c)   # 오래된 것을 지우고
            cache.append(c)   # 최신 것으로 업데이트
            answer += 1
        # cache miss
        else:
            if len(cache) == cacheSize:   # 캐시가 다 찼으면
                cache.pop(0)    # 가장 오래된 것을 삭제
            cache.append(c)
            answer += 5
                
    return answer
```
