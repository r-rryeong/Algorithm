def solution(s):
    answer = []
    
    cnt = 0    # 이진변환 횟수
    zero = 0   # 제거한 0의 개수
    while s != "1":   # 1이 될때까지 반복
        cnt += 1
        len_s1 = len(s)
        s = s.replace("0","")   # 0 제거
        len_s2 = len(s)
        zero += len_s1-len_s2   # 제거한 0의 개수 누적
        
        s = ""   # 이진변환 결과
        while len_s2 != 0:
            q = int(len_s2) % 2
            s += str(q)
            len_s2 = int(len_s2) // 2
        s = s[::-1]   # 거꾸로 들어가므로 뒤집어준다
        
    answer.append(cnt)
    answer.append(zero)
    
    return answer

def binary_num(num):
    result = ''
    
    while num != 0:
        m = num % 2
        result += str(m)
        num //= 2
    
    result = result[::-1]
    
    return result

def solution(s):
    answer = [0, 0]   # 이진 변환의 횟수, 제거된 0의 개수
    
    while s != '1':
        cnt = s.count('0')
        answer[1] += cnt
        s = s.replace('0', '')
        
        num = int(len(s))
        s = binary_num(num)
        answer[0] += 1
    
    return answer
