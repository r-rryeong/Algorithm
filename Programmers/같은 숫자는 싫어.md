from collections import deque

def solution(arr):
    arr = deque(arr)
    answer = []
    
    while arr:
        x = arr.popleft()
        if answer:    # answer에 숫자가 존재하는 경우
            if x == answer[-1]:
                continue
            else:
                answer.append(x)
        else:         # answer에 숫자가 없는 경우
            answer.append(x)
                
    return answer
