def solution(s):
    answer = False
    lst = []
    
    for i in s:
        if i == ')':
            if lst and lst[-1] =='(':
                lst.pop()
            else:
                lst.append(i)
        elif i =='(':
            lst.append(i)
            
    if not lst:
        answer = True

    return answer

from collections import deque

def solution(s):
    answer = True
    path_lst = []
    s = deque(s)
    
    while s:
        p = s.popleft()
        if p == '(':
            path_lst.append(p)
        else:
            if path_lst:
                if path_lst[-1] == '(':
                    path_lst.pop()
            else:
                 return False   
    
    if path_lst:
        answer = False
    
    return answer
