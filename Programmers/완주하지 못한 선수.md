```python
def solution(participant, completion):
    answer = ''
    participant.sort()
    completion.sort()

    i = 0
    j = 0
    while i != len(participant) and j != len(completion):
        if participant[i] == completion[j]:
            i += 1
            j += 1
            # 종료조건(인덱스 초과)
            if i == len(participant) and j == len(completion):
                break
        # 같지 않은 경우 i번째 참가자를 answer에 저장하고 i만 1증가
        else:
            answer = participant[i]
            i += 1

    # 완주하지 못한 선수를 찾지 못하고 빠져나온 경우
    if i == len(participant)-1:
        answer = participant[-1]

    return answer
```

```python
def solution(participant, completion):
    participant.sort()
    completion.sort()
    n = len(participant)
    for i in range(n):
        if i == n-1:
            return participant[i]
        if participant[i] == completion[i]:
            continue
        else:
            return participant[i]
```

```python
def solution(participant, completion):
    answer = ''
    dict = {}   # 참가자 명단을 담은 딕셔너리
    
    for p in participant:
        if p not in dict.keys():   # 아직 딕셔너리에 들어가지 않은 선수라면
            dict[p] = 0   # value 초기화
        dict[p] += 1
        
    for c in completion:
        dict[c] -= 1   # 완주한 선수는 value 값 -1
        
    for i, j in dict.items():   # key, value
        if j:            # value 값이 0보다 크다면
            answer = i   # 완주하지 못한 선수 확인
            break        # for문 종료
    
    return answer
```
