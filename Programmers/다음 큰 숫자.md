def binary_num(n):   # n을 2진수로 변환했을 때 1의 개수
    cnt = 0
    while n > 0:
        if n % 2 == 1:   # 나머지가 1일 때만 count
            cnt += 1
        n //= 2
    
    return cnt

def solution(n):
    answer = n+1
    
    while True:
        if binary_num(n) == binary_num(answer):
            break
        else:
            answer += 1
    
    return answer
