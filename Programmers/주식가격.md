```python
def solution(prices):
    answer = []

    l = len(prices)
    second = 0   # 가격이 떨어지지 않은 초
    for i in range(l-1):
        num = prices[i]   # 기준 값
        for j in range(i+1, l):
            if num <= prices[j]:   # 가격이 떨어지지 않은 경우
                second += 1
            else:         # 가격이 떨어진 경우
                answer.append(second+1)   # 가격이 1초만 유지되었다
                second = 0
                break  # 다음 값 비교를 위해 for문 종료
        if second:     # second가 양수 값이라면
            answer.append(second)
            second = 0

    answer.append(0)   # 마지막 가격은 비교하지 않았으므로 0 추가

    return answer
```
