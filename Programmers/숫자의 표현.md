```python
def solution(n):
    answer = 0
    for i in range(1, n+1):
        total = 0
        for j in range(i, n+1):
            total += j
            if total == n:
                answer += 1
                break
            elif total >= n:
                break
                 
    return answer
```

```python
def solution(n):
    answer = 0
    
    for i in range(1, n+1):
        # n의 약수 중 홀수인 숫자
        if i % 2 and n % i == 0:
            answer += 1
    
    return answer
```
