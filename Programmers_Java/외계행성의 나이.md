class Solution {
    public String solution(int age) {
        String answer = "";
        String str_age = String.valueOf(age);
        String[] lst_age = str_age.split("");
        
        for (int i=0;i<lst_age.length;i++) {
            answer += ((char) (Integer.parseInt(lst_age[i]) + 97));
        }
        
        return answer;
    }
}
